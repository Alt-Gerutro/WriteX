264 try {
265 short cur_filter;
266 {
267 std::lock_guard<std::mutex> lock(mtx);
268 cur_filter = filter_level;
269 }
270 if (static_cast<short>(lvl) & cur_filter) {
271 std::string str = format(lvl, format_msg(msg, std::forward<Args>(args)...), file, func, line);
272 enq_msg(str);
273 }
274 }
275 catch (const std::format_error) {
276 this->log(WriteX_Level::FATAL, "CANNOT FORMAT MESSAGE: {}", file, func, line, msg);
